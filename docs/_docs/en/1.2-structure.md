```bash
├───examples      # notebook examples for every implemented model
├───handyrec
│   ├───dataset   # helper classes for processing data
│   ├───features  # definition of input features
│   ├───layers    # core layers used by models
│   └───models
│       ├───match # implementation of match models
│       └───rank  # implementation of rank models
└───tests         # test code
```
